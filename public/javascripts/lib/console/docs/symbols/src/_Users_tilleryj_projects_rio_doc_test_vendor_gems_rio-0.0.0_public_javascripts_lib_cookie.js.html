1 /**
  2 	@namespace 
  3 	
  4 	Methods for managing cookies.
  5 */
  6 rio.Cookie = {
  7 
  8 	key: "_RIO_",
  9 	bindings: {},
 10 	observers: {},
 11 
 12 	/**
 13 		Gets the value of a cookie.
 14 		
 15 		@param {String} name The name of the cookie to get.
 16 	*/
 17 	get: function(name) {
 18 		var cookies = document.cookie.match(this.key + name + '=(.*?)(;|$)');
 19 		return (cookies) ? unescape(cookies[1]) : null;
 20 	},
 21 
 22 	/**
 23 		Sets the value of a cookie.
 24 		
 25 		@param {String} name The name of the cookie to set.
 26 		@param {Object} value The value to set in the cookie.
 27 	*/
 28 	set: function(name, value) {
 29 		// Set to 5000 days
 30 		var expires = new Date(new Date().getTime()+(1000*60*60*24*5000)).toGMTString();
 31 		var cookie = this.key + name + "=" + escape((value || "").toString());
 32 		try {
 33 			document.cookie = cookie + "; expires=" + expires;
 34 			this["_" + name] = value;
 35 			(this.observers[name] || []).each(function(observer) { observer(value); });
 36 		} catch (e) {
 37 			rio.log("Cookie.js - " + e);
 38 		}
 39 	},
 40 	
 41 	/**
 42 		Set a default value for a cookie if the cookie is not yet defined
 43 		
 44 		@param {String} name The name of the cookie to set.
 45 		@param {Object} value The default value to set in the cookie.
 46 	*/
 47 	setDefault: function(name, value) {
 48 		if (document.cookie.match(this.key + name + '=(.*?)(;|$)') == undefined) {
 49 			this.set(name, value);
 50 		}
 51 	},
 52 	
 53 	bind: function(name, observer) {
 54 		if (this.observers[name] == undefined) { this.observers[name] = []; }
 55 		
 56 		this.observers[name].push(observer);
 57 		observer(this.get(name));
 58 	},
 59 	
 60 	binding: function(name, defaultValue) {
 61 		if (defaultValue) {
 62 			this.setDefault(name, defaultValue);
 63 		}
 64 		if (this.bindings[name]) { return this.bindings[name]; }
 65 		
 66 		this["_" + name] = this.get(name);
 67 		this.bindings[name] = new rio.Binding(this, name);
 68 		
 69 		this[("get-" + name).camelize()] = this.get.curry(name).bind(this);
 70 		this[("set-" + name).camelize()] = this.set.curry(name).bind(this);
 71 		
 72 		return this.bindings[name];
 73 	}
 74 };
