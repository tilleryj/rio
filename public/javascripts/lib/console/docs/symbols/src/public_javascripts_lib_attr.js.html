 96 			fireBinding: function(binding, value) {
 97 				if (Object.isFunction(binding)) {
 98 					// try {
 99 						binding(value);
100 					// } catch(e) {
101 						// TODO: This check is in here because refreshing the console can leave null binding references in the app
102 						// This needs to be resolved immediately.  We should be "un-binding" instead of ignoring this error.
103 					//	rio.log(e);
104 					//}
105 				} else {
106 					if (Object.isArray(value)) {
107 						if (!binding.__unbindings) { binding.__unbindings = []; }
108 						binding.__unbindings.each(function(unbinding) {
109 							unbinding();
110 						});
111 						var oldPush = value.push;
112 						value.push = value.push.wrap(function(proceed, val) {
113 							var ret = proceed(val);
114 							if (binding.insert) { binding.insert(val, value.length - 1); }
115 							if (binding.empty && value.length == 1) { binding.empty(false); }
116 							return ret;
117 						});
118 						
119 						var oldSplice = value.splice;
120 						value.splice = value.splice.wrap(function(proceed, index, count, toInsert) {
121 							var startsEmpty = value.length == 0;
122 
123 							var removing = count > 0 && binding.remove;
124 							var inserting = (toInsert !== undefined) && binding.insert;
125 							if (removing) {	var toRemove = value[index]; }
126 							
127 							var ret = (toInsert !== undefined) ? proceed(index, count, toInsert) : proceed(index, count);
128 
129 							if (removing) { binding.remove(toRemove); }
130 							if (inserting) { binding.insert(toInsert, index); }
131 
132 							var endsEmpty = value.length == 0;
133 							if (binding.empty && (startsEmpty != endsEmpty)) { binding.empty(endsEmpty); }
134 							return ret;
135 						});
136 						
137 						var oldClear = value.clear;
138 						value.clear = value.clear.wrap(function(proceed) {
139 							if (value.length == 0) { return; }
140 							proceed();
141 							if (binding.set) {
142 								binding.set(value);
143 							}
144 						});
145 						
146 						binding.__unbindings.push(function() {
147 							value.push = oldPush;
148 							value.splice = oldSplice;
149 							value.clear = oldClear;
150 						});
151 
152 						if (binding.empty) { binding.empty(value.length == 0); }
153 					}
154 					if (binding.set) { 
155 						binding.set(value);
156 					}
157 				}

164 				
165 				@param {String} attribute The name of the attribute to bind

167 				@param {Boolean} skipInitialExecution (optional) Do <b>not</b> immediately call the function with the attributes current value

170 				attr.bind("attribute", function(newValue) {

174 			bind: function(attribute, binding, skipInitialExecution) {
175 				var unbind;
176 				if (attribute.include(".")) {
177 					var thisAttribute = attribute.match(/^([^\.]*)\.(.*)/)[1];
178 					var subAttributes = attribute.match(/^([^\.]*)\.(.*)/)[2];
179 					if (!this.__subBindings()[thisAttribute]) { this.__subBindings()[thisAttribute] = []; }
180 					this.__subBindings()[thisAttribute].push([subAttributes, binding]);
181 					this.__executeSubBinding(thisAttribute, subAttributes, binding, skipInitialExecution);
182 				} else {
183 					if (!this.__bindings()[attribute]) { this.__bindings()[attribute] = []; }
184 					this.__bindings()[attribute].push(binding);
185 					unbind = this.unbind.curry(attribute, binding).bind(this);
186 					if (!skipInitialExecution) {
187 						this.fireBinding(binding, this[("get-" + attribute).camelize()]());
188 					}
189 				}
190 				return unbind;
191 			},

507 						if (val && val.BINDING) {

510 						}
511 					}

545 /**
546 	@class
547 	
548 	Represents a bindable attribute.  Bindings can be bound to each other to create a bi-directional binding.<br /><br />
549 	
550 	<i>You should rarely need to instantiate this class directly</i>
551 */
552 rio.Binding = Class.create({
553 	BINDING: true,
554 	__bindings: [],
555 	initialize: function(obj, fieldName) {
556 		this._obj = obj;
557 		this._fieldName = fieldName;
558 	},
559 	
560 	value: function() {
561 		if (this._fieldName.include(".")) {
562 			var parts = this._fieldName.split(".");
563 			return parts.inject(this._obj, function(acc, field) {
564 				return acc ? acc[("get-" + field).camelize()]() : null;
565 			}.bind(this));
566 		} else {
567 			return this._obj[("get-" + this._fieldName).camelize()]();
568 		}
569 	},
570 	
571 	bindTo: function(binding) {
572 		this._obj.bind(this._fieldName, binding.update.bind(binding));
573 		binding._obj.bind(binding._fieldName, this.update.bind(this));
574 	},
575 	
576 	bind: function(observer, skipInitialExecution) {
577 		return this._obj.bind(this._fieldName, observer, skipInitialExecution);
578 	},
579 	
580 	update: function(newVal) {
581 		var fieldToSet = this._fieldName;
582 		var setOn = this._obj;
583 		if (this._fieldName.include(".")) {
584 			var parts = this._fieldName.split(".");
585 			fieldToSet = parts.last();
586 			setOn = parts.slice(0, parts.size() - 1).inject(this._obj, function(acc, field) {
587 				return acc ? acc["_" + field] : null;
588 			}.bind(this));
589 		}
590 		if (setOn) {
591 			setOn[("set-" + fieldToSet).camelize()](newVal);
592 		}
593 	}
594 });
