1 /**
  2 	@name rio
  3 	@namespace
  4 	
  5 	The rio namespace is the root of the entire framework, through which you can access all of the classes and 
  6 	namespaces.
  7 	
  8 	<br />
  9 	
 10 	<h2>Namespaces</h2>
 11 	
 12 	<ul>
 13 		<li>apps - contains instances of rio.Application</li>
 14 		<li>components - contains instances of rio.Component</li>
 15 		<li>models - contains instances of rio.Model</li>
 16 		<li>pages - contains instances of rio.Page</li>
 17 		<li>environment - configuration for the current environment</li>
 18 		<li>environments - configurations varying by environment</li>
 19 		<li>console - stuff related to the development console</li>
 20 		<li>boot - configuration related to the application boot process</li>
 21 	</ul>
 22 	
 23 	<h2>Classes</h2>
 24 	
 25 	<ul>
 26 		<li>AIM - manages AJAX file uploads</li>
 27 		<li>Application - root of application instances</li>
 28 		<li>Attr - root of most rio classes (Application, Component, Model, Page)</li>
 29 		<li>Binding - represents a bindable attribute</li>
 30 		<li>Model - ...</li>
 31 		<li>Tag - a collection of methods for building HTML tags</li>
 32 		<li>Utils - a collection of random helper methods</li>
 33 	</ul>
 34 	
 35 	<h2>Methods</h2>
 36 	<ul>
 37 		<li>log - adds a log message to the rio console</li>
 38 	</ul>
 39 	
 40 	<h2>Fields</h2>
 41 	<ul>
 42 		<li>app - the currently running application instance</li>
 43 	</ul>
 44 
 45 	@author Jason Tillery
 46 	@copyright 2008-2009 Thinklink LLC
 47 */
 48 var w = window;
 49 (function() {
 50 	if (!window.console) { window.console = { log: alert }; }
 51 	
 52 	var initialWindowKeys = Object.keys(window);
 53 	
 54 	var bootOptions = window.bootOptions || {};
 55 
 56 	Object.extend(rio, {
 57 		components: {},
 58 		models: {},
 59 		apps: {},
 60 		pages: {},
 61 		console: {},
 62 		themes: {},
 63 		log: Prototype.emptyFunction,
 64 		boot: {
 65 			root: "/javascripts/",
 66 			appRoot: bootOptions.appRoot || "",
 67 			rioPath: bootOptions.rioPath || "lib/",
 68 			prototypePath: bootOptions.prototypePath || "prototype/",
 69 			prototypePollution: [
 70 				"$", "$$", "$A", "$F", "$H", "$R", "$break", "$continue", "$w", "_eventID", "Abstract", "Ajax", "Autocompleter", "Builder", "Class", "Draggable", "Draggables", "Droppables", "Effect", "Element", "Enumerable", "Field", "Form", "Hash", "Insertion", "ObjectRange", "PeriodicalExecuter", "Position", "Prototype", "Selector", "Sortable", "SortableObserver", "Template", "Toggle", "Try"
 71 			],
 72 			googleAnalyticsPollution: ["gaGlobal", "gaJsHost", "_gat", "pageTracker"],
 73 			noConsole: bootOptions.noConsole,
 74 			initialWindowKeys: initialWindowKeys,
 75 			prototypeScripts: [],
 76 			rioScripts: [],
 77 			loadedFiles: [],
 78 			loadFunctions: {},
 79 			
 80 			appScripts: function() {
 81 				return rio.boot.loadedFiles.reject(function(f) {
 82 					return rio.boot.prototypeScripts.include(f) || rio.boot.rioScripts.include(f);
 83 				});
 84 			},
 85 			
 86 			pollution: function() {
 87 				return Object.keys(window).reject(function(key) {
 88 					return rio.boot.initialWindowKeys.include(key) || 
 89 						rio.boot.prototypePollution.include(key) || 
 90 						rio.boot.googleAnalyticsPollution.include(key);
 91 				}.bind(this));
 92 			}
 93 		}
 94 	});
 95 
 96 	$$("script").each(function(script) {
 97 		var src = script.src;
 98 		if(src.match(/boot\.js/) && src.indexOf('?') != -1) {
 99 			var options = src.split('?')[1].split(',');
100 			if (options.length >= 1) { rio.boot.appName = options[0]; }
101 			if (options.length >= 2) { rio.boot.environment = options[1]; }
102 			if (options.length >= 3) { rio.boot.bootCompressed = (options[2] == "compressed"); }
103 		}
104 	});
105 	
106 	Object.extend(rio.environment, rio.environments[rio.boot.environment]);
107 	Object.extend(rio.boot, rio.environment.boot);
108 
109 	if (bootOptions.noConsole === undefined) {
110 		rio.boot.noConsole = (rio.environment.console == undefined) ? false : !rio.environment.console;
111 	}
112 	
113 	rio.require = function(fileName, options) {
114 		options = Object.extend({
115 			force: false,
116 			track: true
117 		}, options || {});
118 
119 		if (!rio.boot.loadedFiles.include(fileName)) {
120 			if (rio.boot.bootCompressed && options.force) {
121 				if (options.track) { rio.boot.loadedFiles.push(fileName); }
122 				document.write('<script type="text/javascript" src="' + rio.boot.root + fileName + '.js"></script>');
123 			} else if (!rio.boot.bootCompressed) {
124 				if (rio.boot.loadFunctions[fileName]) {
125 					rio.boot.loadFunctions[fileName]();
126 				} else {
127 					var path = rio.boot.root + fileName + ".js";
128 					new Ajax.Request(path + "?" + Math.random(), {
129 						asynchronous: false,
130 						method: "get",
131 						evalJSON: false,
132 						evalJS: false,
133 						onSuccess: function(response) {
134 							try {
135 								w.eval(response.responseText);
136 							} catch(e) {
137 								console.log(Object.toJSON(e));
138 							}
139 						},
140 						onFailure: function() {
141 							rio.log("Failed loading file: " + path);
142 						}
143 					});					
144 				}
145 				if (options.track) { rio.boot.loadedFiles.push(fileName); }
146 			}
147 		}
148 	};
149 
150 	if (rio.boot.bootCompressed) {
151 		rio.require(rio.boot.appRoot + "prototype/compressed/prototype", { force: true });
152 		rio.require(rio.boot.rioPath + "compressed/rio", { force: true });
153 		rio.require(rio.boot.appRoot + "apps/compressed/" + rio.boot.appName, { force: true });
154 		if (rio.boot.environment == "development") {
155 			rio.require(rio.boot.rioPath + "rio_development_concat", { force: true });
156 		}
157 	} else {
158 		rio.require(rio.boot.appRoot + "prototype/prototype_concat", { track: false });
159 		rio.require(rio.boot.rioPath + "rio_concat", { track: false });
160 		rio.require(rio.boot.appRoot + "apps/" + rio.boot.appName + "_concat", { track: false });
161 		rio.require(rio.boot.appRoot + "apps/" + rio.boot.appName);
162 	};
163 	
164 	if (rio.environment.supportSelenium) {
165 		rio.require(rio.boot.rioPath + "seleniumExtensions", { force: true });
166 	}
167 	
168 	Event.observe(window, 'load', function() {
169 		window.initialOptions = window.initialOptions || {};
170 		rio.app = new rio.apps[rio.boot.appName.camelize()](initialOptions);
171 		rio.app.launch();
172 	}.bind(this));
173 })();
