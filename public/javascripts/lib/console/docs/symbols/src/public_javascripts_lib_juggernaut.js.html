 24 /* This variable has to be left here because Juggernaut.swf expects it. */
 25 var juggernaut;
 26 
 27 /**
 28 	@class
 29 	
 30 	<p>Juggernaut is a utility that allows the server to initiate connections with the browser.</p>
 31 	<p>This is what facilitates the client-server bindings that allow for automatic-collborative models.</p>
 32 	<p><i>Note: this version of Juggernaut has been altered to work better with the rio framework.</i></p>
 33 */
 34 rio.Juggernaut = Class.create({
 35 	initialize: function(options) {
 36 		this.is_connected = false;
 37 		this.attempting_to_reconnect = false;
 38 		this.ever_been_connected = false;
 39 		this.options = options;
 40 		this.bindToWindow();
 41 		
 42 		var connectionAttr = new rio.Attr.create({
 43 			attrAccessors: [["live", false]]
 44 		});
 45 		this.live = (new connectionAttr()).live;
 46 	},
 47 
 48 	logger: function(msg) {
 49 		if (this.options.debug) {
 50 			msg = "Juggernaut: " + msg + " on " + this.options.host + ':' + this.options.port;
 51 			rio.log(msg);
 52 		}
 53 	},
 54 
 55 	initialized: function(){
 56 		this.fire_event('initialized');
 57 		this.connect();
 58 	},
 59 
 60 	broadcast: function(body, type, client_ids, channels){
 61 		var msg = { command: 'broadcast', body: body, type: (type||'to_channels') };
 62 		if(channels) { msg.channels = channels; }
 63 		if(client_ids) { msg.client_ids = client_ids; }
 64 		msg.client_id = this.options.client_id;
 65 		msg.session_id = this.options.session_id;
 66 		this.sendData(this.toJSON(msg));
 67 	},
 68 
 69 	sendData: function(data){
 70 		this.swf().sendData(escape(data));
 71 	},
 72 
 73 	connect: function(){
 74 		if(!this.is_connected){
 75 			this.fire_event('connect');
 76 			this.swf().connect(this.options.host, this.options.port);
 77 		}
 78 	},
 79 
 80 	disconnect: function(){
 81 		if(this.is_connected) {
 82 			this.swf().disconnect();
 83 			this.is_connected = false;
 84 			this.live.update(false);
 85 		}
 86 	},
 87 	
 88 	// Added to facilitate subscription to new channels
 89 	resetConnection: function() {
 90 		if (!this._resetTask) { this._resetTask = new rio.DelayedTask(); }
 91 		this._resetTask.delay(100, function() {
 92 			if (!(this.swf() && this.swf().connect)) { 
 93 				this.resetConnection.bind(this).delay(500); 
 94 				return;
 95 			}
 96 			this.disconnect();
 97 			this.connect();
 98 		}.bind(this));
 99 	},
100 
101 	handshake: function() {
102 		var handshake = {};
103 		handshake.command = 'subscribe';
104 		if(this.options.session_id) { handshake.session_id = this.options.session_id; }
105 		if(this.options.client_id) { 
106 			handshake.client_id = this.options.client_id; 
107 		} else {
108 			handshake.client_id = this.options.session_id; 
109 		}
110 		if(this.options.channels) { handshake.channels = this.options.channels; }
111 		if(this.currentMsgId) {
112 			handshake.last_msg_id = this.currentMsgId;
113 			handshake.signature = this.currentSignature;
114 		}
115 
116 		return handshake;
117 	},
118 
119 	connected: function(e) {
120 		var json = this.toJSON(this.handshake());
121 		this.sendData(json);
122 		this.ever_been_connected = true;
123 		this.is_connected = true;
124 		this.live.update(true);
125 		setTimeout(function(){
126 			if(this.is_connected) { this.attempting_to_reconnect = false; }
127 		}.bind(this), 1 * 1000);
128 		this.logger('Connected');
129 		this.fire_event('connected');
130 	},
131 
132 	// OVERRIDE FOR CHAT STYLE APPS - POSSIBLE MALICIOUS CONTENT CAN BE EVALED
133 	receiveData: function(e) {
134 		var msg = this.parseJSON(unescape(e.toString()));
135 		this.currentMsgId = msg.id;
136 		this.currentSignature = msg.signature;
137 		
138 		if (msg.body.startsWith("rio.console.touch")) {
139 			// don't log autospec stuff
140 		} else {
141 			this.logger("Received data:\n" + msg.body + "\n");
142 		}
143 		eval(msg.body); 
144 	},
145 
146 	/*** START PROTOTYPE SPECIFIC - OVERRIDE FOR OTHER FRAMEWORKS ***/
147 	fire_event: function(fx_name) {
148 		$(document).fire("juggernaut:" + fx_name);
149 	},
150 
151 	bindToWindow: function() {
152 
153 		Event.observe(window, 'load', function() {      
154 			window.juggernaut = this;
155 			this.appendFlashObject();
156 		}.bind(this));
157 
158 	},
159 
160 	toJSON: function(hash) {
161 		return Object.toJSON(hash);
162 	},
163 
164 	parseJSON: function(string) {
165 		return string.evalJSON();
166 	},
167 
168 	swf: function(){
169 		return $(this.options.swf_name);    
170 	},
171 
172 	appendElement: function() {
173 		this.element = new Element('div', { id: 'juggernaut' });
174 		$(document.body).insert({ bottom: this.element });
175 	},
176 
177 	/*** END PROTOTYPE SPECIFIC ***/
178 
179 	appendFlashObject: function(){
180 		if(this.swf()) {
181 			throw("Juggernaut error. 'swf_name' must be unique per juggernaut instance.");
182 		}
183 		this.appendElement();
184 		rio.swfobject.embedSWF(
185 			this.options.swf_address, 
186 			'juggernaut', 
187 			this.options.width, 
188 			this.options.height, 
189 			String(this.options.flash_version),
190 			this.options.ei_swf_address,
191 			{'bridgeName': this.options.bridge_name},
192 			{},
193 			{'id': this.options.swf_name, 'name': this.options.swf_name}
194 		);
195 	},
196 
197 	refreshFlashObject: function(){
198 		this.swf().remove();
199 		this.appendFlashObject();
200 	},
201 
202 	errorConnecting: function(e) {
203 		this.is_connected = false;
204 		this.live.update(false);
205 		if(!this.attempting_to_reconnect) {
206 			this.logger('There has been an error connecting');
207 			this.fire_event('errorConnecting');
208 			this.reconnect();
209 		}
210 	},
211 
212 	disconnected: function(e) {
213 		this.is_connected = false;
214 		this.live.update(false);
215 		if(!this.attempting_to_reconnect) {
216 			this.logger('Connection has been lost');
217 			this.fire_event('disconnected');
218 			this.reconnect();
219 		}
220 	},
221 
222 	reconnect: function(){
223 		if(this.options.reconnect_attempts){
224 			this.attempting_to_reconnect = true;
225 			this.live.update(false);
226 			this.fire_event('reconnect');
227 			this.logger('Will attempt to reconnect ' + this.options.reconnect_attempts + ' times,' +
228 			'the first in ' + (this.options.reconnect_intervals || 3) + ' seconds');
229 			var attemptReconnect = function(){
230 				if(!this.is_connected){
231 					this.logger('Attempting reconnect');
232 					if(!this.ever_been_connected){
233 						this.refreshFlashObject();
234 					} else {
235 						this.connect();
236 					}
237 				}
238 			}.bind(this);
239 			for(var i=0; i < this.options.reconnect_attempts; i++){
240 				setTimeout(attemptReconnect, (this.options.reconnect_intervals || 3) * 1000 * (i + 1));
241 			}
242 		}
243 	}
244 });
245 
246 rio.Juggernaut.boot = function() {
247 	rio.juggernaut = new rio.Juggernaut({
248 		ei_swf_address: "/javascripts/lib/expressinstall.swf", 
249 		reconnect_intervals: 3, 
250 		flash_version: 8, 
251 		channels: [],
252 		session_id: rio.environment.sessionId,
253 		client_id: rio.environment.userId,
254 		host: rio.environment.juggernautUrl, 
255 		width: "0px", 
256 		height: "0px", 
257 		flash_color: "#fff", 
258 		swf_name: "juggernaut_flash", 
259 		port: rio.environment.juggernautPort, 
260 		bridge_name: "juggernaut", 
261 		debug: rio.environment.juggernautDebug, 
262 		swf_address: "/javascripts/lib/juggernaut.swf", 
263 		reconnect_attempts: 3
264 	});
265 };
