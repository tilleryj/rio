1 /**
  2 	@class
  3 
  4 	Application is used to create new rio application classes.  It provides functionality for dependency management, 
  5 	routing, history management and page management.
  6 	
  7 	@extends rio.Attr
  8 */
  9 rio.Application = {
 10 	create: function() {
 11 		var args = $A(arguments);
 12 		if (args.length > 0 && args.last() != undefined && !args.last().ATTR) {
 13 			args[args.size() - 1] = Object.extend({ noExtend: true }, args.last());
 14 		}
 15 		var app = rio.Attr.create.apply(this, args);
 16 
 17 		app.addMethods({
 18 			initHistory: function() {				
 19 				dhtmlHistory.initialize();
 20 				dhtmlHistory.addListener(this.applyHistoryEntry.bind(this));
 21 			},
 22 			
 23 			applyHistoryEntry: function(location, historyData) {
 24 				if (this.__revertingTransientHistoryEntry) {
 25 					this.addHistoryEntry(this.__revertingTransientHistoryEntry[0], this.__revertingTransientHistoryEntry[1]);
 26 					this.__revertingTransientHistoryEntry = false;
 27 				} else {
 28 					this.navigateTo(location, true);
 29 				}
 30 			},
 31 			
 32 			addHistoryEntry: function(location, transient) {
 33 				if (historyStorage.hasKey(this.getCurrentLocation()) && historyStorage.get(this.getCurrentLocation()).transient) {
 34 					this.__revertingTransientHistoryEntry = [location, transient];
 35 					history.back(1);
 36 				} else {
 37 					dhtmlHistory.add(location, { transient: transient });
 38 				}
 39 			},
 40 			
 41 			resize: function() {
 42 				this.getCurrentPage().resize();
 43 			},
 44 			
 45 			keyPress: function(e) {
 46 				var currentPage = this.getCurrentPage();
 47 				if (currentPage) {
 48 					this.getCurrentPage().keyPress(e);
 49 				}
 50 			},
 51 
 52 			keyDown: function(e) {
 53 				var keyMap = this.getKeyMap();
 54 				if (keyMap) {
 55 					keyMap.handle(e);
 56 				}
 57 				var currentPage = this.getCurrentPage();
 58 				if (currentPage) {
 59 					this.getCurrentPage()._keyDown(e);
 60 				}
 61 			},
 62 			
 63 			unload: function() {
 64 				// meant to be overriden
 65 			},
 66 			
 67 			getKeyMap: function() {
 68 				if (this._keyMap) { return this._keyMap; }
 69 				if (!this.keyMap) { return; }
 70 				
 71 				this._keyMap = rio.KeyMap.build(this.keyMap());
 72 				
 73 				return this._keyMap;
 74 			},
 75 			
 76 			launch: function() {
 77 				document.observe("keypress", this.keyPress.bind(this));
 78 				document.observe("keydown", this.keyDown.bind(this));
 79 				Event.observe(window, "beforeunload", this.unload.bind(this));
 80 
 81 				if (this.noRoutes()) { return; }
 82 
 83 				this.initHistory();
 84 				this.navigateTo(this.getCurrentLocation());
 85 
 86 				Event.observe(window, "resize", this.resize.bind(this));
 87 				
 88 				rio.Application._afterLaunchFunctions.each(function(fcn) {
 89 					fcn(this);
 90 				});
 91 				rio.Application._afterLaunchFunctions.clear();
 92 				
 93 				this._launched = true;
 94 			},
 95 			
 96 			launched: function() {
 97 				return this._launched || false;
 98 			},
 99 			
100 			noRoutes: function() {
101 				return (app.__routes == undefined) || ($H(app.__routes).keys().size() == 0);
102 			},
103 			
104 			avoidAnimation: function() {
105 				return Prototype.Browser.IE;
106 			},
107 			
108 			matchRoutePath: function(path) {
109 				return Object.keys(app.__routes).detect(function(routePath) {
110 					if (routePath == "") { return true; }
111 					var routeParts = routePath.split("/");
112 					var pathParts = path.split("/");
113 					var match = true;
114 					for(var i=0; i<routeParts.length; i++) {
115 						match = match && (routeParts[i].startsWith(":") || routeParts[i].startsWith("*") || routeParts[i] == pathParts[i]);
116 					}
117 					return match;
118 				});
119 			},
120 			
121 			matchRouteTarget: function(path) {
122 				return app.__routes[this.matchRoutePath(path)];
123 			},
124 			
125 			remainingPath: function(path) {
126 				var match = this.matchRoutePath(path);
127 				if (match == "") { return path; }
128 				
129 				var matchParts = match.split("/");
130 				var matchLength = matchParts.last().startsWith("*") ? matchParts.length - 1 : matchParts.length;
131 				
132 				return path.split("/").slice(matchLength).join("/");
133 			},
134 			
135 			pathParts: function(path) {
136 				var parts = {};
137 				var pathParts = path.split("/");
138 				var routePathParts = this.matchRoutePath(path).split("/");
139 				for(var i=0; i<routePathParts.length; i++) {
140 					var routePathPart = routePathParts[i];
141 					if (routePathPart.startsWith(":") || routePathPart.startsWith("*")) {
142 						parts[routePathPart.slice(1)] = pathParts[i];
143 					}
144 				}
145 				return parts;
146 			},
147 			
148 			navigateTo: function(path, noHistoryEntry) {
149 				var subPath = this.matchRoutePath(path);
150 				var target = this.matchRouteTarget(path);
151 				if (!target) { rio.Application.fail("Path not found."); }
152 				
153 				if (path != "" && !noHistoryEntry) {
154 					this.addHistoryEntry(path);
155 				}
156 				var remainingPath = this.remainingPath(path);
157 				var pathParts = this.pathParts(path);
158 				
159 				if (!this.__pages) { this.__pages = {}; }
160 				// Right now the && this.__pages[target] == this.getCurrentPage() prevents double rendering.
161 				var page;
162 				if (this.__pages[target] && this.__pages[target] == this.getCurrentPage()) {
163 					page = this.__pages[target];
164 				} else {
165 					page = this[target]();
166 					this.__pages[target] = page;
167 				}
168 
169 				if (page != this.getCurrentPage()) {
170 					if (this.getCurrentPage()) {
171 						this.clearPage();
172 					}
173 					[Element.body(), Element.html()].each(function(elt) {
174 						elt.setStyle({
175 							width: page.isManagingLayout() ? "100%" : "",
176 							height: page.isManagingLayout() ? "100%" : ""
177 					// 		overflow: page.isManagingLayout() ? "hidden" : ""
178 						});
179 					});
180 					Element.body().insert(page.html());
181 					page.render();
182 					this.setCurrentPage(page);
183 				}
184 
185 				page.applyHistoryEntry(remainingPath, pathParts);
186 			},
187 			
188 			clearPage: function() {
189 				Element.body().childElements().each(function(elt) {
190 					if (elt.tagName.toLowerCase() == 'iframe' || elt.id == 'rshStorageForm' || elt.id == 'rshStorageField' || elt.id == 'juggernaut_flash' || elt.hasClassName("preserve")) {
191 						// keep it
192 					} else {
193 						elt.remove();
194 					}
195 				});
196 			},
197 			
198 			refresh: function() {
199 				document.location.reload();
200 			},
201 			
202 			reboot: function() {
203 				this.clearPage();
204 				this.setCurrentPage(null);
205 				this.navigateTo(this.getCurrentLocation());
206 			},
207 			
208 			getCurrentLocation: function() {
209 				return dhtmlHistory.getCurrentLocation();
210 			},
211 
212 			getCurrentPage: function() {
213 				return this._currentPage;
214 			},
215 
216 			setCurrentPage: function(page) {
217 				this._currentPage = page;
218 			},
219 			
220 			rootUrl: function() {
221 				return document.location.protocol + "//" + document.location.host;
222 			},
223 			
224 			toString: function() {
225 				return "[rio.apps.*]";
226 			}
227 		});
228 		
229 		Object.extend(app, {
230 			route: function(path, target){
231 				if (!this.__routes) { this.__routes = {}; }
232 				this.__routes[path] = target;
233 				
234 				var parts = path.split("/");
235 				if (parts.length > 1) {
236 					for (var i=0; i<parts.length - 1; i++) {
237 						if (parts[i].startsWith("*")) {
238 							throw("Only the final part of a route can be designated as optional by the *");
239 						}
240 					}
241 				}
242 			}
243 		});
244 		
245 		if (args.length > 0 && args.last() != undefined && !args.last().ATTR) {
246 			var initializers = args.last();
247 			if (initializers.requireCss) {
248 				rio.Application.includeCss(initializers.requireCss);
249 			}
250 			Object.keys(initializers.routes || {}).each(function(name) {
251 				app.route(name, initializers.routes[name]);
252 			});
253 			
254 			rio.Application.extend(app, initializers.methods || {});
255 		}
256 
257 		return app;
258 	},
259 	
260 	extend: function(app, extension) {
261 		rio.Attr.extend(app, extension);
262 	},
263 	
264 	include: function(fileName) {
265 		this.require(fileName);
266 		// rio.boot.loadFile(fileName);
267 	},
268 	
269 	require: function(fileName) {
270 		rio.require(fileName);
271 	},
272 
273 
274 	injectCss: function() {
275 		var toLoad = [];
276 		rio.boot.loadedStylesheets.each(function(s) {
277 			if (!rio.preloadedStylesheets.include(s)) { toLoad.push(s); }
278 		});
279 		if (toLoad.empty()) { return; }
280 
281 		var query = toLoad.map(function(f) {
282 			return "files[]=" + f;
283 		}).join("&");
284 		var linkHtml = rio.Tag.link("", {
285 			href: rio.url("/rio/stylesheets?" + query + "&" + rio.cacheKey),
286 			media: "screen",
287 			rel: "stylesheet",
288 			type: "text/css"
289 		});
290 		Element.head().insert({ bottom: linkHtml });
291 	
292 		if (rio.ContainerLayout) {
293 			rio.ContainerLayout.resize();
294 		}	
295 	},
296 
297 	/*
298 		Because of a bug in IE, we need to remove and readd all link tags every time a new one is added.
299 	*/
300 	includeCss: function(toInclude) {
301 		var include = function(fileName) {
302 			if (rio.boot.loadedStylesheets.include(fileName)) { return; }
303 			rio.boot.loadedStylesheets.push(fileName);
304 			if (rio.preloadedStylesheets.include(fileName)) { return; }
305 
306 			if (rio.environment.autoConcatCss && !(rio.app && rio.app.launched())) {
307 				// Do nothing
308 			} else {
309 				var linkHtml = rio.Tag.link("", {
310 					href: rio.url("/stylesheets/" + fileName + ".css"),
311 					media: "screen",
312 					rel: "stylesheet",
313 					type: "text/css"
314 				});
315 				Element.head().insert({ bottom: linkHtml });
316 			}
317 		}.bind(this);
318 		if (Object.isString(toInclude)) {
319 			include(toInclude);
320 		}
321 		if (Object.isArray(toInclude)) {
322 			toInclude.reverse().each(function(css) {
323 				include(css);
324 			});
325 		}
326 	},
327 	
328 	getToken: function() {
329 		return this._token || rio.environment.railsToken;
330 	},
331 
332 	setToken: function(token) {
333 		this._token = token;
334 	},
335 	
336 	_afterLaunchFunctions: [],
337 	afterLaunch: function(afterLoadFunction) {
338 		if (rio.app && rio.app.launched()) { 
339 			afterLoadFunction(rio.app);
340 		} else {
341 			this._afterLaunchFunctions.push(afterLoadFunction);
342 		}
343 	},
344 	
345 	fail: function(msg, description) {
346 		try {
347 			if (rio.app && rio.app.fail) {
348 				rio.app.fail(msg);
349 				rio.error("FAIL: " + msg, description || "", true);
350 			} else {
351 				alert("OOPS: " + msg + "\n\nTry refreshing the page or come back later.");
352 			}
353 		} catch(e) {
354 			// Ignore errors during fail
355 		}
356 	},
357 	
358 	toString: function() {
359 		return "Application";
360 	}
361 };
362 
363 if (!window.dhtmlHistoryCreated) {
364 	window.dhtmlHistory.create({
365 		toJSON: function(o) {
366 			return Object.toJSON(o);
367 		},
368 		fromJSON: function(s) {
369 			return s.evalJSON();
370 		}
371 	});
372 	window.dhtmlHistoryCreated = true;
373 }
