1 /**
  2 	@class
  3 	
  4 	Templates are used to parse rio templates (*.jst files).  Templates are created by rio pages as a starting point
  5 	for a pages html state.  You should rarely have to instantiate this class directly.
  6 */
  7 rio.Template = Class.create({
  8 	initialize: function(options) {
  9 		this._name = options.name || "template";
 10 		this._template = options.text;
 11 	},
 12 	
 13 	mixinMethods: function() {
 14 		var elementMethods = {};
 15 		this.rioIds().each(function(id) {
 16 			var newId = "rio_" + id;
 17 			var name = (id + "Element");
 18 			var fieldName = "_" + name;
 19 			var methodName = ("get-" + name).camelize();
 20 			
 21 			elementMethods[methodName] = function() {
 22 				if (!this[fieldName]) { this[fieldName] = $(newId); }
 23 				return this[fieldName];
 24 			};
 25 		});
 26 		
 27 		var template = this;
 28 		elementMethods.buildHtml = function() {
 29 			var html = rio.Tag.div("", {style: "height: 100%;"}); 
 30 			html.innerHTML = template.render(this);
 31 			return html;
 32 		};
 33 
 34 		return elementMethods;
 35 	},
 36 	
 37 	rioIds: function() {
 38 		return (this._template.match(/rio:id=\"[^\"]*\"/g) || []).map(function(m) {
 39 			return m.match(/rio:id=\"([^\"]*)\"/)[1];
 40 		});
 41 	},
 42 	
 43 	render: function() {
 44 		return this._template.gsub(/rio:id=\"([^\"]*)\"/, function(matches) {
 45 			var id = matches[1];
 46 			var newId = "rio_" + id;
 47 			
 48 			return "id=\"" + newId + "\"";
 49 		});
 50 	},
 51 	
 52 	name: function() {
 53 		return this._name;
 54 	}
 55 });
 56 rio.Template.toString = function() { return "Template"; };
